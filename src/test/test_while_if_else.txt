method test(x: Int, y: Int)
    ensures forall _s: State :: <_s> implies get(_s, y) == 1
{

    assume x > 0

    while (x > 0)
        invariant forall _s: State :: <_s> implies get(_s, x) >= 0 && (get(_s, x) == 0 implies get(_s, y) == 1)
    {
        if (x == 1) {
            y := 1
        } else {
            y := 2
        }
        x := x - 1
    }

}
