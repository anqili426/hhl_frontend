method test(x: Int) returns (y: Int)
    ensures forall _s: State :: <_s> ==> get(_s, y) == 1
{
    assume x > 0

    var i: Int
    i := x


    while (i > 0)
        invariant forall _s: State :: <_s> ==> get(_s, i) >= 0 && (get(_s, i) == 0 ==> get(_s, y) == 1)
    {
        if (i == 1) {
            y := 1
        } else {
            y := 2
        }
        i := i - 1
    }

}
