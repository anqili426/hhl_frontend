method infBranchTS_GNI(low: Int, high: Int) returns (res: Int)
ensures forall _s1: State, _s2: State :: <_s1> && <_s2> ==>
    (exists _s: State :: <_s> && get(_s, high) == get(_s1, high)
                                && get(_s, res) == get(_s2, res)
                                && get(_s, low) == get(_s2, low)
    )
{
    var x: Int
    if (high > 0) {
        while (x > 0) {
            var y: Int
            havoc y
            if (y > 1) {
                res := x - y
            } else {

}
